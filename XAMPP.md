# Disinstallazione
	
	sudo /opt/lampp/lampp stop
	sudo rm -rf /opt/lampp
	sudo rm -rf /usr/local/bin/lampp
	ls /opt/lampp

# Comandi

	sudo /opt/lampp/lampp start
	sudo /opt/lampp/lampp stop
	sudo /opt/lampp/lampp restart
	sudo /opt/lampp/lampp reload
	sudo /opt/lampp/lampp status
